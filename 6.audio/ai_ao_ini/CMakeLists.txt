
project(ai_ao_ini)

# link_directories(${EXECUTABLE_OUTPUT_PATH})

    "${CMAKE_COMMAND} -E copy_directory ${CMAKE_CURRENT_SOURCE_DIR}/wav ${EXECUTABLE_OUTPUT_PATH}/wav/"
